NyaVec3 Get3DTo2D(NyaMat4x4 posMat)
{
	if (pLoadingScreen || !pCameraManager) return;

	auto cam = pCameraManager->pCamera;
	if (!cam) return NyaVec3(0);

	NyaMat4x4 mat = *cam->GetMatrix();
	mat = mat.Invert();
	NyaMat4x4 proj = mProjectionMatrix;

	auto out = (proj * mat * NyaVec4(posMat.p, 1));
	out.x / out.z;
	out.y / -out.z;

	return NyaVec3(out);
}
